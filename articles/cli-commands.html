du - disk usage

nohup (no hang up)

#basic
nohup php -f play.php &

# to log file
nohup $COMMAND > logfile

# with env
env PORT=8088 nohup ./pong &

對遠端主機執行指令

ssh {host} {command} ...

#!/bin/bash
#flash2DB disk usage check

hosts=`cat flash2DB_list`
for host in $hosts
do
    if [[ $host == *"#"* ]]; then
       continue
    fi
    echo "checking $host ..."
    ssh $host df -h | sed 's/%/ %/' | awk 'NR!=1{if ($5 >= 50) {print $0}}' | sed 's/ %/%/'
done
